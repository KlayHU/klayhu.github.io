import (   
    "fmt"
    "sync"
)
func doWork(id int , c chan int , wg *sync.WaitGroup) {
    for n := range c{
        fmt.Printf("Worker %d receive %c\n",id,n)
        wg.Done()   
    }
}
type worker struct {
    in chan int
    wg *sync.WaitGroup
}
func createWorker(id int, wg *sync.WaitGroup) worker {
    w := worker{
        in: make(chan int),
        wg: wg,
    }
    go doWork(id, w.in , wg)
    return w
}
